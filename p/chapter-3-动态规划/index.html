void MatrixChainOrder(int p[], int n){
    int dp[n][n];
    for(int i = 1; i < n; i++) dp[i][i] = 0;

    for(int l = 2; l < n; l++){ // l is chain length
        for(int i = 1; i < n - l + 1; i++){
            int j = i + l - 1;
            dp[i][j] = INT_MAX;
            for(int k = i; k <= j - 1; k++){
                int q = dp[i][k] + dp[k+1][j] + p[i-1]*p[k]*p[j];
                if(q < dp[i][j]) dp[i][j] = q;
            }
        }
    }
}

void MAXSUB(int a[], int n){
    int *b = new int[MAX];
    int *rec = new int[max];
    b[0] = a[0];
    rec[0] = 0;

    int max = b[0];
    
    for(int i = 1; i < n; i++){
        if(b[i-1] > 0) {
            b[i] = b[i-1] + a[i];
            rec = rec[i-1];
        }
        else{
            b[i] = a[i];
            rec = i;
        }
        if(b[i] > max) max = b[i];
    }
}

void LCS(char X[], char Y[], int m, int n){
    int dp[m+1][n+1];
    for(int i = 0; i <= m; i++) dp[i][0] = 0;
    for(int j = 0; j <= n; j++) dp[0][j] = 0;

    for(int i = 1; i <= m; i++){
        for(int j = 1; j <= n; j++){
            if(X[i-1] == Y[j-1]){
                dp[i][j] = dp[i-1][j-1] + 1;
            }
            else{
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
            }
        }
    }
}

void Knapsack(int w[], int v[], int n, int W){
    int dp[n+1][W+1];
    for(int i = 0; i <= n; i++) dp[i][0] = 0;
    for(int j = 0; j <= W; j++) dp[0][j] = 0;

    for(int i = 1; i <= n; i++){
        for(int j = 1; j <= W; j++){
            if(j < w[i-1]){
                dp[i][j] = dp[i-1][j];
            }
            else{
                dp[i][j] = max(dp[i-1][j], dp[i-1][j-w[i-1]] + v[i-1]);
            }
        }
    }
}
